from collections import Counter

# inside class SearchOrchestrator:

    def _apply_gates(
        self,
        scored: List[ScoredResult],
        plan: QueryPlan,
        project_meta: Dict[str, Dict],
    ) -> List[ScoredResult]:
        """
        Hard filters applied after scoring but before dedup and top_k slicing.

        Only uses existing meta fields:
        - is_screenshot, face_count, width, height, has_gps, flag
        """
        if not scored:
            return scored

        preset = (plan.preset_id or "").strip().lower()
        filters = plan.filters or {}

        # Infer gate intent from preset_id and filters only.
        require_screenshot = (preset in {"screenshots", "screenshot"}) or bool(filters.get("is_screenshot") is True)
        exclude_screenshots = (preset in {"documents", "document"}) or bool(filters.get("exclude_screenshots") is True)

        require_gps = bool(filters.get("has_gps") is True) or bool(filters.get("has_location") is True)
        require_favorite = (preset in {"favorites", "favourites"}) or (filters.get("flag") == "pick")

        # Documents should default to excluding faces even if caller forgot to set exclude_faces
        exclude_faces = bool(plan.exclude_faces) or (preset in {"documents", "document"}) or bool(filters.get("exclude_faces") is True)

        # Optional: reject tiny images for docs (still meta-only). Tune later if needed.
        min_doc_edge = 350 if preset in {"documents", "document"} else None

        dropped = Counter()
        kept: List[ScoredResult] = []

        for r in scored:
            meta = project_meta.get(r.path, {})

            is_screenshot = bool(meta.get("is_screenshot", False))
            face_count = int(meta.get("face_count") or 0)
            has_gps = bool(meta.get("has_gps", False))
            flag = meta.get("flag", None)

            w = meta.get("width")
            h = meta.get("height")
            try:
                w = int(w) if w is not None else None
                h = int(h) if h is not None else None
            except Exception:
                w, h = None, None

            # Gate: require screenshot
            if require_screenshot and not is_screenshot:
                dropped["require_screenshot"] += 1
                continue

            # Gate: exclude screenshots
            if exclude_screenshots and is_screenshot:
                dropped["exclude_screenshot"] += 1
                continue

            # Gate: require GPS
            if require_gps and not has_gps:
                dropped["require_gps"] += 1
                continue

            # Gate: require favorite flag
            if require_favorite and flag != "pick":
                dropped["require_favorite"] += 1
                continue

            # Gate: exclude faces
            if exclude_faces and face_count > 0:
                dropped["exclude_faces"] += 1
                continue

            # Gate: documents, drop tiny images
            if min_doc_edge is not None and w is not None and h is not None:
                if min(w, h) < min_doc_edge:
                    dropped["doc_too_small"] += 1
                    continue

            kept.append(r)

        if dropped:
            logger.info(
                f"[SearchOrchestrator] Gates applied preset={plan.preset_id!r} "
                f"kept={len(kept)}/{len(scored)} dropped={dict(dropped)}"
            )

        return kept
