```python
def _is_strong_type_asset(
    self,
    path: str,
    project_meta: Dict[str, Dict],
) -> bool:
    meta = project_meta.get(path, {}) or {}
    ext = os.path.splitext(path)[1].lower() if path else ""
    ocr_text = (meta.get("ocr_text") or "").strip()
    ocr_len = len(ocr_text)
    is_screenshot = bool(meta.get("is_screenshot"))
    face_count = int(meta.get("face_count") or 0)

    if is_screenshot:
        return True

    if ext in {".pdf", ".tif", ".tiff"}:
        return True

    if ext == ".png" and ocr_len >= 20 and face_count == 0:
        return True

    if ocr_len >= 60 and face_count == 0:
        return True

    return False
```

### Patch 3.4, add scenic hard-prune helper

```python
def _filter_scenic_candidates(
    self,
    candidate_paths: List[str],
    project_meta: Dict[str, Dict],
) -> List[str]:
    kept = []
    dropped = {
        "strong_type_asset": 0,
    }

    for path in candidate_paths:
        if self._is_strong_type_asset(path, project_meta):
            dropped["strong_type_asset"] += 1
            continue
        kept.append(path)

    logger.info(
        f"[SearchOrchestrator] Scenic hard-prune: kept={len(kept)} "
        f"dropped={dropped}"
    )
    return kept
```

### Patch 4.4, add `_count_soft_evidence()`

```python
@staticmethod
def _count_soft_evidence(
    results: list,
    candidate_set: CandidateSet,
    family: str,
) -> int:
    """Count weaker, provisional evidence."""
    count = 0
    top_n = min(20, len(results))

    for r in results[:top_n]:
        path = r.path if hasattr(r, "path") else r
        evidence = candidate_set.evidence_by_path.get(path, {})

        if family == "documents":
            if evidence.get("provisional_document"):
                count += 1

    return count
```
